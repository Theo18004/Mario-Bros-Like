132 // -- Calcul de la position sur l'écran (Destination) --
133 // Le sprite est souvent plus grand que la hitbox physique.
134 // On centre le sprite horizontalement sur la hitbox.
135 // On aligne le bas du sprite avec le bas de la hitbox (les pieds).
136
137 // On peut grossir le sprite (scale) si nécessaire. Ici on garde la taille originale (x1.5 par exemple)
138 float scale = 1.5f;
139 int displayW = (int)(frameWidth * scale);
140 int displayH = (int)(texH * scale);
141
142 SDL_Rect destRect = {
143 .x = player.rect.x - (displayW - player.rect.w) / 2, // Centrage horizontal
144 .y = player.rect.y - (displayH - player.rect.h), // Alignement par le bas
145 .w = displayW,
146 .h = displayH
147 };
148
149 // Gestion de la direction (Flip)
150 SDL_RendererFlip flip = player.facingLeft ? SDL_FLIP_HORIZONTAL : SDL_FLIP_NONE;
151
152 // Dessin final
153 SDL_RenderCopyEx(renderer, currentTex, &srcRect, &destRect, 0.0, NULL, flip);
154 }

165 if (player.texJump) SDL_DestroyTexture(player.texJump);
166 if (mapBackground) SDL_DestroyTexture(mapBackground);
167
168 SDL_DestroyRenderer(renderer);
169 SDL_DestroyWindow(window);
